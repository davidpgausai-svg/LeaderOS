  function renderLevels() {
    const container = document.getElementById('levelsContainer');
    container.innerHTML = '';

    levels.forEach(level => {
      const card = document.createElement('div');
      card.className = `level-card level-${level.number}`;
      card.id = `level-${level.number}`;

      let contentHTML = `
        <div class="level-header" onclick="toggleLevel(${level.number})">
          <div class="level-title">
            <div class="level-number">${level.number}</div>
            <div class="level-info">
              <h3>${level.emoji} Level ${level.number} ‚Äî ${level.title}</h3>
              <p>Mindset: "${level.mindset}"</p>

        </div>
        <div class="level-content">
          <p style="line-height: 1.6; color: #374151; margin-bottom: 16px;">${level.description}</p>
      `;

      if (level.prompts) {
        contentHTML += `
          <div class="prompt-examples">
            <h4>üí° Prompts to Try:</h4>
            ${level.prompts.map(prompt => `
              <div class="prompt-item" onclick="copyPrompt(this)">${prompt}</div>
            `).join('')}
          </div>
        `;
      }

      if (level.framework) {
        contentHTML += `
          <div class="framework-box">
            <div class="framework-title">üß≠ Framework: ${level.framework.name}</div>
            <div class="framework-content">
        `;

        if (level.framework.example) {
          contentHTML += `<p style="margin: 8px 0 0 0;">"${level.framework.example}"</p>`;
        }

        if (level.framework.steps) {
          contentHTML += '<ol style="margin-top: 8px;">';

            <strong>üéØ Coaching Tip:</strong>

          </div>
        </div>
      `;

      card.innerHTML = contentHTML;
      container.appendChild(card);
    });
  }

  function toggleLevel(number) {
    const card = document.getElementById(`level-${number}`);
    card.classList.toggle('open');
  }

  let allOpen = false;
  function toggleAll() {
    allOpen = !allOpen;
    const cards = document.querySelectorAll('.level-card');
    const button = document.querySelector('.open-all button');
    
    cards.forEach(card => {
      if (allOpen) {
        card.classList.add('open');
      } else {
        card.classList.remove('open');
      }
    });

    button.textContent = allOpen ? 'Collapse All Levels' : 'Expand All Levels';
  }

  function copyPrompt(element) {
    const text = element.textContent;
    navigator.clipboard.writeText(text).then(() => {
      const original = element.style.background;
      element.style.background = '#d1fae5';
      setTimeout(() => {
        element.style.background = original;
      }, 500);
    });
  }
